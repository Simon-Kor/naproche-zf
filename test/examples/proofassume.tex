\begin{proposition}\label{assumetest}
    If $x\in y$, then $x\in y$.
\end{proposition}
\begin{proof}
    Assume $x\in y$.
    Then $x\in y$.
\end{proof}

\begin{proposition}\label{assumetesttwo}
    If $x\in y$ and $a\in b$, then $x\in y$.
\end{proposition}
\begin{proof}
    Assume $a\in b$.
    Assume $x\in y$.
    Then $x\in y$.
\end{proof}
