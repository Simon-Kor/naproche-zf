\import{function.tex}
\import{nat.tex}

\begin{definition}\label{subfinite}
    $X$ is subfinite iff
        there exists a natural number $k$ such that
        there exists an injective function $f$ from $k$ to $X$.
\end{definition}

\begin{proposition}\label{emptyset_subfinite}
    $\emptyset$ is subfinite.
\end{proposition}
